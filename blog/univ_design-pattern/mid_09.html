  Object[] data;
  int position = 0;

  public MyListIterator(Object[] data) {
    this.data = data;
  }

  public Object next() {
    Object obj = data[position];
    position += 1;
    return obj;
  }

  public boolean hasNext() {
    if (position >= data.length || data[position] == null) {
      return false;
    }
    return true;
  }
}
